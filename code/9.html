break语句

break的作用在于跳出当前循环（注意是当前循环而不是所有的循环）且能跳出swiitch的后续判断
栗子1：

        int main(){
            int n=0;
            while(1){//死循环1
                while(1){//死循环2
                    n++;
                    printf("循环第%d次\n",n);//每次运行输出一次
                   break;//当输出一次时跳出循环2
                }
                if(n==3){
                    break;//当n==3时跳出循环1
                }
            }
        return 0;
        }

栗子2： 
        int main(){
            int i=1;
            switch(i){
                case 1:printf("1反馈");
                case 2:printf("2反馈");break;//当为1时两个值具输出代表break的跳出对switch语句极其重要。
            }
        return 0;
        }

continue跳出

continue跳出与break存在差别，其作用并不是跳出整个循环而是跳出当前循环
栗子： 

        int main(){
            int n;
            for(n=1;n<10;n++){
                if(n==4){//当n=4时跳出当前循环
                    continue;
                }
                printf("%d\n",n);
            }
            return 0;
        }

你可以发现输出中缺少4的输出，因为continue在n==4时跳出了当前循环进入了下一循环，所以不存在输出。

goto语句

goto语句还是C语言中比较有意思的语句其能在程序中标记一个要跳回的地方然后在另一个地方标记一个if判断的跳回点或跳出点来回跳

不建议在较大的程序中使用goto语句因为他会严重降低程序的可读性以及后续维护的难度 
        int main(){
            int n=1;
            loop://标记点
            printf("%d",n)
            if(n!=10){
                n++;
                goto loop;//跳回到标记点
            }//该方法使用goto与if建立一个循环程序
        }

循环的嵌套：示例99乘法算术表的打印

        #include<stdio.h>
        #include<stdlib.h>//调用system语句
        int main(){
            int i,j;
            for(i=1;i<10;i++){
                for(j=1;j<=i;j++){//仅输出不重复的部分
                    printf("%dx%d=%d\t",j,i,i*j);//\为制表符用于平衡间距类似于excel表格
                }
                printf("\n");//当每输出一次时换行
            }
            system("pause");//system可用于调用命令行来完成部分功能此处指使程序在结束后停止需要一次输入后继续（相当于暂停）
            return 0;
